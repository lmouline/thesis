\newcommand*\cleartoevenpage{
  \clearpage

}

% Cover definition
\newcommand{\prvBackCover}[1]
{
	\nolinenumbers
	\cleartoevenpage
	\includepdf{#1}
}

\newcommand{\frontCover}[1]
{
	\ifthenelse{\equal{\setCover}{\coverHal}}
		{\includepdf{cover/Cover.pdf}}
		{
			\ifthenelse{\equal{\setCover}{\coverLux}}
			{\includepdf{cover/CoverLux.pdf}}
			{\includepdf{cover/CoverRennes.pdf}}
		
		}
}

\newcommand{\backCover}[1]
{
	\ifthenelse{\equal{\setCover}{\coverRennes}}
		{\prvBackCover{cover/BackCoverRennes.pdf}}
		{\prvBackCover{cover/BackCoverLuxembourg.pdf}}
}
